datos<-read.csv("movies.csv")

#Normalizar datos
datos2<-datos[complete.cases(datos),]

#Seleccionar variables a trabajar
popularidad<-datos2[,'popularity']
presupuesto<-datos2[,'budget']
ingresos<-datos2[,'revenue']
duracion<-datos2[,'runtime']
votos<-datos2[,'voteCount']
#Creamos el dataframe con los datos a utilizar
datosCluster<-data.frame(popularidad,presupuesto,ingresos,duracion,votos)
#Estadistico de hopkins
hopkins(datosCluster)
## [1] 0.9998335
#VAT
datos_dist<- dist(datosCluster)
knitr::opts_chunk$set(fig.width=12, fig.height=8)
